\section{Basic Programming Concepts}

\subsection{Variables}
\begin{frame}[fragile]
\frametitle{Variables}
\begin{itemize}
    \item Variables are used to store data.
    \item Example in C\#:
\end{itemize}
\begin{lstlisting}
int x = 5;
string name = "John";
\end{lstlisting}
\end{frame}

\subsection{Data Types}
\begin{frame}[fragile]
\frametitle{Data Types}
\begin{itemize}
    \item Data types define the type of data a variable can hold.
    \item Example in C\#:
\end{itemize}
\begin{lstlisting}
int age = 30;
double salary = 50000.50;
bool isEmployed = true;
\end{lstlisting}
\end{frame}

\subsection{Control Flow}
\begin{frame}[fragile]
\frametitle{Control Flow}
\begin{itemize}
    \item Control flow statements control the order in which instructions are executed.
    \item Example in C\#:
\end{itemitem}
\begin{lstlisting}
if (age > 18) {
    Console.WriteLine("Adult");
} else {
    Console.WriteLine("Minor");
}
\end{lstlisting}
\end{frame}
